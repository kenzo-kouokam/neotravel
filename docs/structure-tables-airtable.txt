Légende des types :
  autoNumber          = identifiant auto-incrémenté (clé primaire)
  singleLineText      = texte court
  multilineText       = texte long
  email / phoneNumber = champs typés
  number              = nombre
  currency            = montant (€)
  percent             = pourcentage (0.15 = 15 %)
  date                = date
  createdTime         = date de création (auto, lecture seule)
  lastModifiedTime    = date de dernière modification (auto, lecture seule)
  formula             = champ calculé (lecture seule)
  singleSelect        = liste déroulante (un choix)
  multipleSelects     = liste déroulante (plusieurs choix)
  multipleRecordLinks = lien vers une autre table
